We can create a word cloud highlighting the names of passengers.

import matplotlib.pyplot as plt
from wordcloud import WordCloud

# Here I process the names by removing repeated words other than the names
names = ' '.join(data.Name.values)
words = names.replace(',', '')\
     .replace('Mr.', '')\
     .replace('Mrs.', '')\
     .replace('Miss.', '')\
     .replace('Master.', '')\
     .replace('Dr.', '')

# Create the cloud
wc = WordCloud(width=1000, height=800, background_color='white').generate(words)
plt.figure(figsize=(12, 8))
plt.imshow(wc)
plt.axis('off')
plt.show()
